const [currentPlayer, setCurrentPlayer] = useState(1);
  const [playerScores, setPlayerScores] = useState([0, 0, 0]);
  const [activeQuestion, setActiveQuestion] = useState(null);

  const handlePlayerClick = (playerNum) => {
    // only allow player clicks if there is an active question
    if (activeQuestion) {
      setCurrentPlayer(playerNum); //changes the current player to that number if there is an active question. 
                                     If there is no active question, the player click is ignored.
    }
  };

  const handleAnswer = (answer) => {
    const isCorrect = answer === activeQuestion.answer; //function takes an answer as input and checks 
                                                          if it is correct by comparing it to the answer of the active question
    const newScores = [...playerScores];

    if (isCorrect) {
      newScores[currentPlayer - 1] += 1;
      setPlayerScores(newScores);
    } else {
      // if both other players answer incorrectly, switch to the next player
      if (currentPlayer === 1) {
        setCurrentPlayer(2);
      } else if (currentPlayer === 2) {
        setCurrentPlayer(3);
      } else {
        setActiveQuestion(null);
      }
    }
  };
